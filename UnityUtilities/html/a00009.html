UnityHelper

The UnityHelper contains a plethora of useful extensions and helpers for Transform, GameObject, Vector2/3/4, Rect and more.

Examples

Transform/Vector/Color

1 {C#}
2 /* Set the transform.position.x to 5 and z to 3. Keeps y.
3  * Equivalent to:
4  * var position = transform.position;
5  * position.x = 5;
6  * position.z = 3;
7  * transform.position = position;
8  */
9 transform.SetPosition(x: 5, z: 3);
10 
11 // Same as above; only sets transform.localEulerAngles.y.
12 // There are extension methods for all position/rotation/scales.
13 transform.SetLocalEulerAngles(y: 180);
14 
15 // Similar methods are available for Vector2/3/4 and Color:
16 // Gets the transform.position, but with y set to 0.
17 Vector3 floorPosition = transform.position.Change3(y: 0);
18 
19 // Gets the material color, but sets the color.a value to 0.5.
20 Color halfTransparentColor = GetComponent<Renderer>().sharedMaterial.color.ChangeAlpha(0.5f);
21 
22 // Sets the position/rotation of enemyIndicator to someEnemyTransform.position/rotation
23 enemyIndicator.CopyPositionAndRotatationFrom(someEnemyTransform);

Vector2 Rotation

1 {C#}
2 // Create a length 1 Vector2 pointing 40 degrees away from (1.0, 0.0)
3 var vector = UnityHelper.CreateVector2AngleDeg(20f);
4 Debug.Log(vector); // => (0.9, 0.3)
5 
6 // Rotate the vector 70 degrees
7 vector = vector.RotateDeg(70f);
8 Debug.Log(vector); // => (0.0, 1.0)
9 
10 // Output the current vector rotation
11 Debug.Log(vector.GetAngleDeg()); // => 90

Rect

1 {C#}
2 // Make a rect from (10|20) to (60|120)
3 Rect rect = new Rect(10, 20, 50, 100);
4 
5 // Gets a random position for an enemy in the rect, leaving a 5 unit border
6 Vector2 enemySpawnPosition = rect.RandomPosition(-5);
7 
8 // Gets a random sub rect of size 10|10 in which we could spawn multiple enemies

10 
11 Vector2 enemyPosition = new Vector2(0, 500);
12 
13 // Clamp an enemy position to the rect
14 enemyPosition = rect.Clamp2(enemyPosition);
15 Debug.Log(enemyPosition); // Output: (10.0, 120.0)
16 
17 // Create a rect that is 10 units bigger to each side
18 Rect biggerRect = rect.Extend(10);
19 
20 // Get the corner points
