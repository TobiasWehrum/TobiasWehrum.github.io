Countdown

This class is useful for things like cooldowns or spawn delays. It is also helpful for tweening things by using the PercentElapsed property.

(If you want to tween a lot of stuff, you might want to check out DOTween. Actually, check it out either way, DOTween is fabulous!)

Examples

CountdownExample

1 {C#}
2 public class CountdownExample : MonoBehaviour
3 {
4  Countdown countdownLoop;
5  Countdown countdownRandom;
6  Countdown countdownOnce;
7  Countdown countdownDelay;
8 
9  void Awake()
10  {
11  // Looping countdown, every 2 seconds
12  countdownLoop = new Countdown(true, 2f);
13 
14  // Looping countdown, delegate is called every loop to determine duration (1 to 5 seconds)
15  countdownRandom = new Countdown(true, () => UnityEngine.Random.Range(1f, 5f));
16 
17  // Non-looping countdown, 5 seconds duration. Can be reset by calling Reset().
18  countdownOnce = new Countdown(false, 5f);
19 
20  // Non-looping countdown that isn't started yet. Can be started by calling e.g. Reset(5f).
21  countdownDelay = new Countdown(false);
22  }
23 
24  void Update()
25  {
26  // Looping automatically
27  if (countdownLoop.Progress())
28  Debug.Log("This is logged every 2 seconds.");
29 
30  // Looping automatically. Duration is chosen between 1 and 5 every loop.
31  if (countdownRandom.Progress())
32  Debug.Log("This is logged every 1 to 5 seconds.");
33 
34  // Not looping
35  if (countdownOnce.Progress())
36  Debug.Log("This is shown once after 5 seconds.");
37 
38  // Start countdownDelay (again) - if it's already running, it is reset
39  if (Input.GetKeyDown(KeyCode.Alpha1))
40  countdownDelay.Reset(1f); // 1 second
41 
42  if (Input.GetKeyDown(KeyCode.Alpha2))
43  countdownDelay.Reset(2f); // 2 seconds
44 
45  // Show this once 1/2 seconds after 1/2 was last pressed
46  if (countdownDelay.Progress())
47  Debug.Log("This is shown after " + countdownDelay.Duration + " seconds.");
48 
49  // Output data about the countdown when space is pressed
50  if (Input.GetKeyDown(KeyCode.Space))
51  {
52  if (countdownDelay.IsRunning)
53  Debug.Log("countdownDelay is " + (countdownDelay.PercentElapsed * 100) + "% complete.");
54  else
55  Debug.Log("countdownDelay is stopped.");
56  }
57  }

CountdownPropertyExample

You can also use CountdownPropertyOnce and CountdownPropertyLoop to get the values directly from the editor. Live-editing works - if you change the value in the editor, it will be used in the next loop.

1 {C#}
2 public class CountdownPropertyExample : MonoBehaviour
3 {
4  [SerializeField] CountdownPropertyOnce countdownOnce;
5  [SerializeField] CountdownPropertyLoop countdownLoop;
6 
7  void Update()
8  {
9  // Looping automatically
10  if (countdownLoop.Progress())
11  Debug.Log("This is logged every " + countdownLoop.Duration + " seconds.");
12 
13  // Not looping
14  if (countdownOnce.Progress())
15  Debug.Log("This is shown once after " + countdownOnce.Duration + " seconds.");
16 
17  // Start countdownOnce again - if it's already running, it is reset
18  if (Input.GetKeyDown(KeyCode.Space))
19  countdownOnce.Reset();
20  }
21 }
