XmlHelper

The XmlHelper serializes and deserializes to/from XML and allows convenient access to optional element content and attributes when reading general XMLs.

With a few exceptions (e.g. arrays of ArrayList and arrays of List<T>), all public attributes and fields of any public class should be serialized without any further need to tag the elements. The only thing needed is a public default constructor.

For finer control, see the MSDN documentation on XmlSerializer: https://msdn.microsoft.com/en-us/library/system.xml.serialization.xmlserializer(v=vs.110).aspx

String serialization

1 {C#}
2 public class TestData
3 {
4  // A public field - will be serialized
5  public int field;
6 
7  // A private field with public property - will be serialized
8  float property;
9  public float Property
10  {
11  get { return property; }
12  set { property = value;}
13  }
14 
15  // An auto property - will be serialized
16  public bool AutoProperty { get; set; }
17 
18  // A private field - will *not* be serialized
19  string privateField = "Test";
20 
21  // A public field marked "XmlIgnore" - will *not* be serialized
22  [XmlIgnore]
23  public double publicNonSerialized = 5.5;
24 
25  // The public default constructor is needed for the XmlSerializer.
26  public TestData()
27  {
28  }
29 
30  public TestData(int field, float property, bool autoProperty)
31  {
32  this.field = field;
33  this.property = property;
34  AutoProperty = autoProperty;
35  }
36 }
1 {C#}
2 // Create a new TestData object
3 TestData data = new TestData(1, 2.3f, true);
4 
5 // Serialize the TestData object into a string
6 string xmlString = data.SerializeToXmlString();
7 
8 /* Output:
9 
10 <?xml version="1.0" encoding="utf-16"?>
11 <TestData xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns:xsd="http://www.w3.org/2001/XMLSchema">
12  <field>1</field>
13  <Property>2.3</Property>
14  <AutoProperty>true</AutoProperty>
15 </TestData>
16  */
17 Debug.Log(xmlString);
18 
19 // Get the data back from the string
20 TestData deserializedData = xmlString.DeserializeFromXmlString<TestData>();

XmlNode content/attributes

1 {C#}
2 // Create an XmlDocument with test data
3 XmlDocument xmlDocument = new XmlDocument();
4 xmlDocument.LoadXml("<enemyList>" +
5  " <enemyData>" +
6  " <name>Grunt</name>" +
7  " <position x='5' y='3'/>" +
8  " </enemyData>" +
9  " <enemyData>" +
10  " <name>Tank</name>" +
11  " <position x='7' y='1'/>" +
12  " <ranged>true</ranged>" +
13  " </enemyData>" +
14  "</enemyList>");
15 
16 // Read each enemyData element in the enemyList
17 foreach (XmlNode enemyData in xmlDocument["enemyList"].ChildNodes)
18 {
19  // Get the name element content, if it exists, else set "???"
20  string name = enemyData.GetElementString("name", "???");
21 
22  // Get the position element and then its attributes
23  XmlNode position = enemyData["position"];
24  int x = position.GetAttributeInt("x");
25  int y = position.GetAttributeInt("y");
26 
27  // Get the ranged element content, if it exists, else set "false"
28  bool ranged = enemyData.GetElementBool("ranged", false);
29 
30  // Output the result
31  Debug.Log(string.Format("{0} at {1}|{2} is {3}",
32  name,
33  x,
34  y,
35  ranged ? "ranged" : "not ranged"));
36 }
37 
38 /* Grunt at 5|3 is not ranged
39  Tank at 7|1 is ranged
40  */
